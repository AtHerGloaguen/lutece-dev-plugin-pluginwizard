		DAOUtil daoUtil = new DAOUtil( SQL_QUERY_NEW_PK  );
		daoUtil.executeQuery();

		if( !daoUtil.next() )

			// if the table is empty
			nKey = 1;
		}

		nKey = daoUtil.getInt( 1 ) + 1;
		daoUtil.free();

		return nKey;
	}
